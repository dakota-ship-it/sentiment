rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Helper function to check if user is authenticated
    function isAuthenticated() {
      return request.auth != null;
    }

    // Optional: Helper function to check if user has AdClass email
    // Uncomment to restrict access to only AdClass users
    // function isAdClassUser() {
    //   return request.auth != null &&
    //          request.auth.token.email.matches('.*@adclass[.].*');
    // }

    match /clients/{clientId} {
      allow read, write, create, delete: if isAuthenticated();
    }

    match /analyses/{analysisId} {
      allow read, write, create, delete: if isAuthenticated();
    }

    match /transcript_queues/{clientId} {
      allow read, write, create, delete: if isAuthenticated();
    }

    match /client_mappings/{clientId} {
      allow read, write, create, delete: if isAuthenticated();
    }

    match /notification_preferences/{clientId} {
      allow read, write, create, delete: if isAuthenticated();
    }
  }
}
